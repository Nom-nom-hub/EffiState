// Creating and using a store is incredibly simple
import { createStore } from 'effistate';

// Initialize with any state object
const store = createStore({
  count: 0,
  todos: [],
  user: { name: 'Guest' }
});

// Get the current state
const state = store.getAll();

// Update state (partial updates work)
store.set({ count: state.count + 1 });

// Subscribe to changes
store.subscribe((newState, oldState) => {
  console.log('State changed:', newState);
});

// Vue 3 Composition API integration
import { createStore } from 'effistate';
import { createVueBindings } from 'effistate/vue';

// Create your store
const store = createStore({
  products: [],
  cart: []
});

// Create Vue bindings
const { useState, useSelector, createStorePlugin } = createVueBindings(store);

// Use in your components
export default {
  setup() {
    // Get reactive state
    const [state, setState] = useState();
    
    // Or select specific parts of state
    const products = useSelector('products');
    const cartCount = useSelector(state => state.cart.length);
    
    // Add methods that update state
    const addToCart = (product) => {
      setState({
        cart: [...state.value.cart, product]
      });
    };
    
    return {
      products,
      cartCount,
      addToCart
    };
  }
}
// Advanced features for power users
import { createStore, withServerSync } from 'effistate';

// Create store with advanced options
const store = createStore(initialState, {
  enableDevTools: true,
  immutable: true,
  enableStructuralSharing: true,
  historyLimit: 50
});

// Enable time-travel debugging
const history = store.enableHistory();
history.undo(); // Go back in time
history.redo(); // Go forward in time

// Specialized array operations (2-3x faster)
store.array.push('todos', newTodo);
store.array.filter('todos', todo => !todo.completed);
store.array.map('todos', todo => ({ ...todo, modified: true }));

// Create optimized selectors
const incompleteTodos = store.createSelector(
  state => state.todos.filter(todo => !todo.completed),
  ['todos'] // Only recompute when todos change
);

// Enhanced with server synchronization
const syncedStore = withServerSync(store, {
  fetchFn: async () => (await fetch('/api/state')).json(),
  pushFn: async (state) => fetch('/api/state', {
    method: 'POST',
    body: JSON.stringify(state)
  }),
  syncInterval: 30000 // Auto-sync every 30 seconds
});
